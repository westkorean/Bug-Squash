        Q1:
            Fix: Missing a ":" in your namespace resolution operator. This results in a compiler error.

            #include <iostream>

            int main() {
                std::cout << "Hello world!";
                return 0;
            }

        Q2:
            Fix: If only 5 expected, change <= 5 to < 5 (6 vs 5 iterations).

            #include <iostream>
            int main(){
                for (int i = 0; i < 5; i++) {
                    std::cout << i << " ";
                }
                return 0;
            }

            std::cout << *ptr;
